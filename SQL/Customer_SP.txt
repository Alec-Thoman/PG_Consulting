Create PROCEDURE [dbo].[Customer_SP]
	-- Parameters added here
	@UserName as NVARCHAR(50),
	@PassWord as NVARCHAR(50)
AS
BEGIN

	Set NOCOUNT ON;

	SELECT Username, Password
	FROM CustomerUserInfo
	WHERE Username = @UserName
	AND Password = @PassWord;

END
